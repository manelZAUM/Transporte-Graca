// ========================================== // LISTA MESTRA DE ADMINISTRADORES // ========================================== var EMAILS_AUTORIZADOS = [ "manoelfariaspr@gmail.com", "coordenador@gmail.com" ]; // ========================================== // MAPEAMENTO EXATO DAS SUAS COLUNAS (Baseado nas imagens) // ========================================== var MAP = { nome: 3, // Coluna D cpf: 6, // Coluna G instituicao: 9, // Coluna J turno: 10, // Coluna K dias: 11, // Coluna L embarque: 12, // Coluna M status: 16, // Coluna Q onibus: 17 // Coluna R }; // ========================================== // FUNÇÃO GET: LÊ OS DADOS E FAZ LOGIN // ========================================== function doGet(e) { try { if (!e || !e.parameter) { return retornarJSON({ erro: "Parâmetros ausentes na requisição." }); } var acao = e.parameter.acao; var tokenParam = e.parameter.token; if (acao === "validar_admin") { var emailReal = decodificarEmailDoToken(tokenParam); if (emailReal && emailReal.toLowerCase().endsWith("@gmail.com") && EMAILS_AUTORIZADOS.includes(emailReal.toLowerCase())) { return retornarJSON({ autorizado: true }); } else { return retornarJSON({ erro: "Acesso negado. Apenas e-mails autorizados têm acesso." }); } } // Acessando a planilha com o seu ID exato var sheet = SpreadsheetApp.openById('1X2nqYNhn4Iz0-id5AhMO0Rgjhw7qxKQx237oaBP9CHY').getSheets()[0]; var data = sheet.getDataRange().getDisplayValues(); if (data.length <= 1) return retornarJSON({ erro: "A base de dados está vazia." }); var alunos = []; // Loop de leitura da planilha for (var i = 1; i < data.length; i++) { var row = data[i]; if (!row[MAP.nome] && !row[MAP.cpf]) continue; alunos.push({ nome: row[MAP.nome] || "", cpf: row[MAP.cpf] || "", instituicao: row[MAP.instituicao] || "", turno: row[MAP.turno] || "", dias: row[MAP.dias] || "", // Lendo a coluna L embarque: row[MAP.embarque] || "", // Lendo a coluna M onibus: row[MAP.onibus] || "", status: row[MAP.status] || "Pendente" }); } // Bloco de validação de acesso corrigido if (acao === "painel_admin" || acao === "dashboard") { if (acao === "painel_admin") { var emailAdmin = decodificarEmailDoToken(tokenParam); if (emailAdmin && EMAILS_AUTORIZADOS.includes(emailAdmin.toLowerCase())) { return retornarJSON({ sucesso: true, dadosGerais: alunos }); } else { return retornarJSON({ erro: "Tentativa de invasão bloqueada. Token inválido." }); } } else if (acao === "dashboard") { var cpfDashboard = e.parameter.cpf; if (!cpfDashboard) { return retornarJSON({ erro: "CPF não informado." }); } var cpfLimpoDashboard = cpfDashboard.replace(/\D/g, ''); var alunoDashboard = alunos.filter(function(a) { return (a.cpf || "").toString().replace(/\D/g, '') === cpfLimpoDashboard; }); if (alunoDashboard.length > 0) { // Retorna os dados completos do aluno, incluindo os novos mapeamentos return retornarJSON({ sucesso: true, dadosGerais: alunoDashboard }); } else { return retornarJSON({ erro: "CPF não encontrado. Procure a Secretaria." }); } } } var cpfParam = e.parameter.cpf; var rotaRequisitada = e.parameter.rota; if (!cpfParam) return retornarJSON({ erro: "CPF não informado." }); var cpfLimpo = cpfParam.replace(/\D/g, ''); var alunoEncontrado = alunos.find(function(a) { return (a.cpf || "").toString().replace(/\D/g, '') === cpfLimpo; }); if (!alunoEncontrado) return retornarJSON({ erro: "CPF não encontrado na base de dados." }); function descobrirOnibus(aluno) { var inst = (aluno.instituicao || "").toUpperCase(); var onibusPlanilha = (aluno.onibus || "").toUpperCase(); if (onibusPlanilha.indexOf('AMARELO') !== -1) return "AMARELO"; if (onibusPlanilha.indexOf('AZUL') !== -1) return "AZUL"; var vaiAmarelo = ["UNINTA", "FLF", "LUCIANO FEIJÃO", "ANHANGUERA", "GRAU", "UVA-CCS"].some(function(facul) { return inst.indexOf(facul) !== -1; }); return vaiAmarelo ? "AMARELO" : "AZUL"; } if (acao === "login_rota") { if (alunoEncontrado.status !== "Aprovado" && alunoEncontrado.status !== "Prioridade") { return retornarJSON({ erro: "Acesso negado: O seu cadastro consta como '" + alunoEncontrado.status + "'." }); } var onibusReal = descobrirOnibus(alunoEncontrado); if (rotaRequisitada && onibusReal !== rotaRequisitada.toUpperCase()) { return retornarJSON({ erro: "Acesso negado: O seu autocarro é o " + onibusReal + "." }); } var passageirosSeguros = alunos.filter(function(a) { return (a.status === "Aprovado" || a.status === "Prioridade") && descobrirOnibus(a) === rotaRequisitada.toUpperCase(); }).map(function(a) { return { nome: a.nome, instituicao: a.instituicao, turno: a.turno, dias: a.dias, embarque: a.embarque }; }); return retornarJSON({ sucesso: true, dados: alunoEncontrado, passageiros: passageirosSeguros }); } return retornarJSON({ erro: "Ação inválida." }); } catch(erroInterno) { return retornarJSON({ erro: "Erro no servidor (doGet): " + erroInterno.message }); } } // ========================================== // FUNÇÃO POST: ESCREVE E EDITA OS DADOS (CRUD) // ========================================== function doPost(e) { try { if (!e || !e.postData || !e.postData.contents) { return retornarJSON({ erro: "Requisição inválida ou sem corpo de dados." }); } var payload = JSON.parse(e.postData.contents); var acao = payload.acao; var token = payload.token; var emailAdmin = decodificarEmailDoToken(token); if (!emailAdmin || !EMAILS_AUTORIZADOS.includes(emailAdmin.toLowerCase())) { return retornarJSON({ erro: "Acesso negado para edição. A sua sessão expirou." }); } var sheet = SpreadsheetApp.openById('1X2nqYNhn4Iz0-id5AhMO0Rgjhw7qxKQx237oaBP9CHY').getSheets()[0]; if (sheet.getMaxColumns() < 18) { sheet.insertColumnsAfter(sheet.getMaxColumns(), 18 - sheet.getMaxColumns()); } if (acao === "adicionar_aluno") { var novaLinha = new Array(18).fill(""); novaLinha[0] = Utilities.getUuid(); novaLinha[1] = Utilities.formatDate(new Date(), "GMT-3", "dd/MM/yyyy HH:mm:ss"); novaLinha[MAP.nome] = payload.nome; novaLinha[MAP.cpf] = payload.cpf; novaLinha[MAP.instituicao] = payload.instituicao; novaLinha[MAP.turno] = payload.turno; novaLinha[MAP.onibus] = payload.onibus; novaLinha[MAP.status] = payload.status; if (payload.dias) novaLinha[MAP.dias] = payload.dias; if (payload.embarque) novaLinha[MAP.embarque] = payload.embarque; sheet.appendRow(novaLinha); return retornarJSON({ sucesso: true }); } if (acao === "editar_aluno" || acao === "excluir_aluno") { var cpfAlvo = (acao === "editar_aluno") ? payload.cpfAntigo : payload.cpf; var cpfLimpoAlvo = cpfAlvo.replace(/\D/g, ''); var values = sheet.getDataRange().getDisplayValues(); var linhaEncontrada = -1; for(var i = 1; i < values.length; i++) { var cpfNaPlanilha = values[i][MAP.cpf] ? values[i][MAP.cpf].toString().replace(/\D/g, '') : ""; if (cpfNaPlanilha === cpfLimpoAlvo) { linhaEncontrada = i + 1; break; } } if (linhaEncontrada === -1) { return retornarJSON({ erro: "Aluno não encontrado na base de dados (CPF não localizou linha)." }); } if (acao === "excluir_aluno") { sheet.deleteRow(linhaEncontrada); return retornarJSON({ sucesso: true }); } if (acao === "editar_aluno") { sheet.getRange(linhaEncontrada, MAP.nome + 1).setValue(payload.nome); sheet.getRange(linhaEncontrada, MAP.cpf + 1).setValue(payload.cpf); sheet.getRange(linhaEncontrada, MAP.instituicao + 1).setValue(payload.instituicao); sheet.getRange(linhaEncontrada, MAP.turno + 1).setValue(payload.turno); sheet.getRange(linhaEncontrada, MAP.onibus + 1).setValue(payload.onibus); sheet.getRange(linhaEncontrada, MAP.status + 1).setValue(payload.status); if (payload.dias !== undefined) sheet.getRange(linhaEncontrada, MAP.dias + 1).setValue(payload.dias); if (payload.embarque !== undefined) sheet.getRange(linhaEncontrada, MAP.embarque + 1).setValue(payload.embarque); return retornarJSON({ sucesso: true }); } } } catch(erro) { return retornarJSON({ erro: "Erro interno no servidor (doPost): " + erro.message }); } } // ========================================== // FUNÇÕES AUXILIARES // ========================================== function retornarJSON(dados) { return ContentService.createTextOutput(JSON.stringify(dados)).setMimeType(ContentService.MimeType.JSON); } function decodificarEmailDoToken(token) { try { if (!token) return null; var partes = token.split('.'); if (partes.length !== 3) return null; var base64 = partes[1].replace(/-/g, '+').replace(/_/g, '/'); while (base64.length % 4 !== 0) { base64 += '='; } var payload = Utilities.base64Decode(base64); return JSON.parse(Utilities.newBlob(payload).getDataAsString()).email; } catch (erro) { return null; } }
